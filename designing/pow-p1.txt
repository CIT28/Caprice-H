<-- schema -->
CREATE TABLE IF NOT EXISTS "riders" (
    "id" INTEGER,
    "name" TEXT,
    PRIMARY KEY ("id")

);
CREATE TABLE IF NOT EXISTS "stations" (
    "id" INTEGER,
    "name" TEXT,
    "line" TEXT,
    PRIMARY KEY ("id")

);
CREATE TABLE IF NOT EXISTS "visits" (
    "id" INTEGER,
    "rider_id" INTEGER,
    "station_id" INTEGER,
    PRIMARY KEY ("id"),
    FOREIGN KEY ("rider_id") REFERENCES "riders"("id"),
    FOREIGN KEY ("station_id") REFERENCES "stations"("id")

);
